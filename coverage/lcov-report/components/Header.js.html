import React, { useState } from "react";
import { Link } from "react-router-dom";
import useOnline from "../utils/useOnline";
import { useSelector } from "react-redux";
 
function Header() {
  const [loginText, setLoginText] = useState("Login");
 
  const online = useOnline();
 
  const cartItems = useSelector((store) => store.cart.items);
  console.log(cartItems);
 
  return (
    <div className="flex justify-between h-14 items-center bg-green-400 shadow-lg px-16">
      <div className="font-bold text-white">
        <h3>Logo</h3>
      </div>
      <div>
        <ul className="flex justify-between items-center gap-10 text-white">
          <li>
            <Link to="/">Home</Link>
          </li>
          <li>
            <Link to="/about">About</Link>
          </li>
          <li>
            <Link to="/contact">Contact</Link>
          </li>
          <li>
            <Link to="/cart">Cart - ({cartItems.length})</Link>
          </li>
          <li>
            <Link to="/grocery">Grocery</Link>
          </li>
          <button
            onClick={() =>
              loginText == "Login"
                ? setLoginText("Logout")
                : setLoginText("Login")
            }
            className="border-2 px-2 rounded-2xl bg-red-500 py-1"
          >
            {loginText}
          </button>
          <div>
            {online ? <p>Online</p> : <p className="text-red-600">Offline</p>}
          </div>
        </ul>
      </div>
    </div>
  );
}
 
export default Header;
